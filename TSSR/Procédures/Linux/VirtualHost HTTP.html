VirtualHost HTTP

<VirtualHost *:80>

        ServerName wordpress
        ServerAdmin webmaster@localhost
        DocumentRoot /var/www/wordpress

        <Directory /var/www/wordpress>
        Options -Indexes
        AllowOverride all
        Order allow,deny
        allow from all
        </Directory>

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

        ServerSignature Off

</VirtualHost>
