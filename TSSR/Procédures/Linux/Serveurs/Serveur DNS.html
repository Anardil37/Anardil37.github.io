Serveur DNS

apt install bind9 dnsutils

 

nano /etc/bind/named.conf

zone “tssr.local”{
	type master;
	file “/var/cache/bind/db.domain”;
};

zone “240.168.192.in-addr.arpa”{
	type master;
	file “/var/cache/bind/db.ip”
};

 

cp /etc/bind/db.local /var/cache/bind/db.domain
cp /etc/bind/db.127 /var/cache/bind/db.ip

 

nano /var/cache/bind/db.domain

nano /var/cache/bind/db.ip

nano /etc/resolv.conf

nameserver	192.168.240.132
systemctl bind9 restart
dig tssr.local
